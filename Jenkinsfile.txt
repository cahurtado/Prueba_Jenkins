pipeline {
	agent any
    environment {
		PROJECT_NAME='KAOS - SOFTWARE PROJECT'
    }
    stages {
		stage('Etapa 0') {
			steps {
				echo "Proyecto ${PROJECT_NAME}"
			}
		}
        stage('Etapa 1 - Validation') {
            steps {
                bat "mvn -version"
                bat "mvn validate"
            }
        }
		stage('Etapa 2 - Compile') {
            steps {
                script{
                    def COMPILE = bat (script:"mvn compile", returnStdout: true)
                    emailext (
                        subject: "Project compiled ${PROJECT_NAME}",
                        body: "${COMPILE}",
                        to: "camilohz9315@gmail.com",
                        from: "camilohz9315@gmail.com"
                    )
                }
            }
        }
        stage('Etapa 3 - Test') {
            steps {
                script{
                    def TEST = bat (script:"mvn test", returnStdout: true)
                    emailext (
                        subject: "Test results ${PROJECT_NAME}",
                        body: "${TEST}",
                        to: "camilohz9315@gmail.com",
                        from: "camilohz9315@gmail.com"
                    )
                }
            }
        }
        stage('Etapa 4 - Package') {
            steps {
                bat "mvn package"
            }
        }
        stage('Etapa 5 - Key') {
            steps {
                script{
                    def JAR = findFiles(glob: '**/target/*.jar')
                    bat "certutil -hashfile ${JAR[0].path} SHA256 | findstr /v hash > huella_mvn.txt"
                }
                
            }
        }
        stage('Etapa 6 - Distribution Preproduction') {
            steps {
                script{
                    def JAR = findFiles(glob: '**/target/*.jar')
                    bat("move /Y ${WORKSPACE}\\${JAR[0].path} F:\\Preproduction")
                    bat("xcopy huella_mvn.txt F:\\Preproduction /Y")
                }
            }
        }
        stage('Etapa 7 - Distribution Production') {
            steps {
                bat("xcopy ${WORKSPACE} F:\\Production /E /S /T /H /Y")
            }
        }
    }
	post {
		always {
            echo "Finalizo: ${currentBuild.currentResult}"
        }
        aborted {
            echo "BUILD ABORTED"
        }
        success {
            echo "BUILD SUCCESS"
            echo "Seria bueno mantener esta construccion"
        }
        unstable {
            echo "BUILD UNSTABLE"
        }
        failure {
            echo "BUILD FAILURE"
			echo "Revisar Fallos"
        }
	}
}
