pipeline {
	agent any
    environment {
		PROJECT_NAME='SCM_CALI - VIDEO'
    }
    stages {
		stage('Etapa 0') {
			steps {
				echo "Proyecto ${PROJECT_NAME}"
			}
		}
        stage('Etapa 1 - Maven Validate') {
            steps {
                bat "mvn -version"
                bat "mvn validate"
            }
        }
		stage('Etapa 12 - Maven Compile') {
            steps {
                script{
                    def COMPILE = bat (script:"mvn compile", returnStdout: true)
                    emailext (
                        subject: "Se compilÃ³ el proyecto",
                        body: "${COMPILE}",
                        to: "camilo323@hotmail.com",
                        from: "camilohz9315@gmail.com"
                    )
                }
            }
        }
        stage('Etapa 3 - Maven Test') {
            steps {
                bat "mvn test"
            }
        }
        stage('Etapa 4 - Maven Test') {
            steps {
                bat "mvn package"
            }
        }
        stage('Etapa 5 - Crear key certutil') {
            steps {
                script{
                    def files = findFiles(glob: '**/target/*.jar')
                    bat "certutil -hashfile ${files[0].path} SHA256 | findstr /v hash > huella_mvn.txt"
                }
            }
        }
    }
	post {
		always {
            echo "Finalizo: ${currentBuild.currentResult}"
        }
        aborted {
            echo "BUILD ABORTED"
        }
        success {
            echo "BUILD SUCCESS"
            echo "Seria bueno mantener esta construccion"
        }
        unstable {
            echo "BUILD UNSTABLE"
        }
        failure {

        }
	}
}
