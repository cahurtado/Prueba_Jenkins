pipeline {
	agent any
    environment {
		PROJECT_NAME='SCM_CALI - VIDEO'
    }
    stages {
		stage('Etapa 0') {
			steps {
				echo "Proyecto ${PROJECT_NAME}"
			}
		}
		stage('Etapa 1 - Maven') {
            steps {
                bat "mvn -version"
                bat "mvn install"
                bat "mvn validate"
                bat "mvn compile"
                bat "mvn package"
            }
        }
        stage('Etapa 2 - Crear key certutil') {
            steps {
                script{
                    def files = findFiles(glob: '**/target/*.jar')
                    bat "certutil -hashfile ${files[0].path} SHA256 | findstr /v hash > huella_mvn.txt"
                    echo "${files[0].path}"
                }
            }
        }
    }
	post {
		always {
            echo "Finalizo: ${currentBuild.currentResult}"
        }
        aborted {
            echo "BUILD ABORTED"
        }
        success {
            echo "BUILD SUCCESS"
            echo "Seria bueno mantener esta construccion"
        }
        unstable {
            echo "BUILD UNSTABLE"
        }
        failure {
            echo "BUILD FAILURE"
			echo "Revisar Fallos"
        }
	}
}
