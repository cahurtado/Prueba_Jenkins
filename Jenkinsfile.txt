pipeline {
	agent any
    environment {
		PROJECT_NAME='SCM_CALI - VIDEO'
    }
    stages {
		stage('Etapa 0') {
			steps {
				echo "Proyecto ${PROJECT_NAME}"
			}
		}
		stage('Etapa 1 - Maven') {
            steps {
                bat "mvn -version"
                bat "mvn clean install"
            }
        }
    }
	post {
		always {
            echo "Finalizo: ${currentBuild.currentResult}"
        }
        aborted {
            echo "BUILD ABORTED"
        }
        success {
            echo "BUILD SUCCESS"
            echo "Seria bueno mantener esta construccion"
        }
        unstable {
            echo "BUILD UNSTABLE"
        }
        failure {
            echo "BUILD FAILURE"
			echo "Revisar Fallos"
        }
	}
}
