pipeline {
	agent any
    environment {
		PROJECT_NAME='SCM_CALI - VIDEO'
    }
    stages {
		stage('Etapa 0') {
			steps {
				echo "Proyecto ${PROJECT_NAME}"
			}
		}
		stage('Etapa 1 - Maven') {
            steps {
                bat "mvn -version"
                bat "mvn install"
                bat "mvn validate"
                bat "mvn compile"
                bat "mvn package"
            }
        }
        stage('Etapa 2 - Crear key certutil') {
            steps {
                @for %F in (target/*.jar) do @certutil -hashfile target/"%F" SHA256 | findstr /v hash > huella_mvn.txt
            }
        }
    }
	post {
		always {
            echo "Finalizo: ${currentBuild.currentResult}"
        }
        aborted {
            echo "BUILD ABORTED"
        }
        success {
            echo "BUILD SUCCESS"
            echo "Seria bueno mantener esta construccion"
        }
        unstable {
            echo "BUILD UNSTABLE"
        }
        failure {
            echo "BUILD FAILURE"
			echo "Revisar Fallos"
        }
	}
}
